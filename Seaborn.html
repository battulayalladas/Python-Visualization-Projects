# Line Plot
import seaborn as sns
import pandas as pd
import matplotlib.pyplot as plt

df = pd.DataFrame({
    "Month": ["Jan", "Feb", "Mar", "Apr", "May"],
    "Sales": [100, 120, 90, 150, 200]
})

sns.lineplot(data=df, x="Month", y="Sales")

plt.title("Sales Over Time")
plt.show()

#Basic Histogram

import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

df = pd.DataFrame({
    "age": [18, 20, 22, 22, 23, 24, 24, 25, 30, 35, 35, 40]
})

sns.histplot(data=df, x="age", bins=5, kde=True)

plt.title("Age Distribution")
plt.show()
